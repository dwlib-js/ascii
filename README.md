# ASCII

## Install
`npm i --save @dwlib/ascii`

## Usage
```javascript
// CJS
const ASCII = require('@dwlib/ascii');
const ASCIIDecode = require('@dwlib/ascii/decode');
const ASCIIEncode = require('@dwlib/ascii/encode');
const ASCIIEncodeInto = require('@dwlib/ascii/encodeInto');
const DecodeString = require('@dwlib/ascii/abstract/DecodeString');
const EncodeString = require('@dwlib/ascii/abstract/EncodeString');
const EncodeStringOrThrow = require('@dwlib/ascii/abstract/EncodeStringOrThrow');
const EncodeStringInto = require('@dwlib/ascii/abstract/EncodeStringInto');
const EncodeStringIntoOrThrow = require('@dwlib/ascii/abstract/EncodeStringIntoOrThrow');
const ThrowInvalidCharacterError = require('@dwlib/ascii/abstract/ThrowInvalidCharacterError');
// ESM
import ASCII from '@dwlib/ascii';
import ASCIIDecode from '@dwlib/ascii/decode';
import ASCIIEncode from '@dwlib/ascii/encode';
import ASCIIEncodeInto from '@dwlib/ascii/encodeInto';
import DecodeString from '@dwlib/ascii/abstract/DecodeString';
import EncodeString from '@dwlib/ascii/abstract/EncodeString';
import EncodeStringOrThrow from '@dwlib/ascii/abstract/EncodeStringOrThrow';
import EncodeStringInto from '@dwlib/ascii/abstract/EncodeStringInto';
import EncodeStringIntoOrThrow from '@dwlib/ascii/abstract/EncodeStringIntoOrThrow';
import ThrowInvalidCharacterError from '@dwlib/ascii/abstract/ThrowInvalidCharacterError';
```
